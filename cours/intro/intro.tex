\begin{frame}{Objectifs}
Objectifs du cours:
\begin{itemize}
\item Aperçu \textit{rapide} de la sécurité informatique 
\item Principales problématiques de la sécurité informatique
\item Avoir quelques bases pour comprendre des argumentaires techniques et les discours
commerciaux relatifs 
\item Se former (et former les autres) aux bons usages
\end{itemize}
\end{frame}      

\begin{frame}{Découpage des cours}
4 parties:
\begin{itemize}
\item La politque de sécurité dans l'entreprise
\item Cryptologie
\begin{itemize}
\item Chiffrement symétrique
\item Chiffrement asymétrique
\item Signature et certification
\end{itemize}
\item Sécurité logicielle
\begin{itemize}
\item Le reverse engineering
\item Les failles logicielles et leur exploitation
\item Logiciels malveillants et contre-mesures
\end{itemize}
\item Sécurité réseau (?)
\begin{itemize}
\item Vulnérabilités des protocoles
\item Quelques attaques célèbres
\item Détection et prévention
\end{itemize}
\end{itemize}
\end{frame}
