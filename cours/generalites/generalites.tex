\section{Généralités}
\subsection{Définition et enjeux}


\begin{frame}{Définition de la sécurité informatique}
\begin{definition}
Information security is the protection of information [Assets]
from a wide range of threats in order to ensure business
continuity, minimize business risks and maximize return on
investment and business opportunities.
\end{definition}
\end{frame}

\begin{frame}{Sécurité informatique}
\begin{itemize}
\item Les systèmes informatiques sont au coeur
des systèmes d'information
\item Ils sont devenus la cible de ceux qui
convoitent l'information
\item Assurer la sécurité de l'information implique
d'assurer la sécurité des systèmes
informatiques
\end{itemize}
\end{frame}

\begin{frame}{Qui sont les pirates ?}
\begin{itemize}
\item Peut être n'importe qui avec l'évolution et la vulgarisation des connaissances
\item Beaucoup d'outils sont disponibles sur Internet
\item Trois générations :
\begin{itemize}
\item 80's-90's: techniciens éclairés
\item 1990-2000: script kiddies
\item 2000-aujourd'hui: l'insécurité devient facilement rentable 
\begin{itemize}
\item Pub, SPAM
\item e-commerce
\item Espionnage industriel
\end{itemize}
\end{itemize}
\end{itemize}
\end{frame}

\begin{frame}{Quels sont leurs objectifs}
Objectifs des attaques:
\begin{itemize}
\item Prouver ses compétences techniques
\item Représailles
\item Désinformer
\item Empêcher l'accès à une ressource (Bombes logiques, DOS)
\item Prendre le contrôle d'une ressource (BotNets)
\item Récupérer de l'information sur un système (Espionnage industriel)
\item Générer des revenus: (Vol, Extorsion, Publicité)
\end{itemize}
\end{frame}

\begin{frame}{Les risques pour l'entreprise}
Les risques encourus par l'entreprise:
\begin{itemize}
\item Les risque stratégiques
\begin{itemize}
\item Destruction/altération de données
\begin{itemize}
\item Exemple: \texttt{Boeing} (57000\$ après une intrusion)
\end{itemize}
\item Vol de données stratégiques
\begin{itemize}
\item Exemple: \texttt{Gartner William Malik} (900 M\$)
\end{itemize}
\end{itemize}
\item Les autres risques
\begin{itemize}
\item Le commerce electronique
\begin{itemize}
\item Exemple: \texttt{Kevin Mitnick} et \texttt{Netcom} 
\end{itemize}
\item Piratage de site web
\begin{itemize}
\item Image de marque, militantisme
\end{itemize}
\end{itemize}
\end{itemize}
\end{frame}

\begin{frame}{Confidentialité}
\begin{definition}
Propriété d'une donnée dont la diffusion doit être limitée aux
seules personnes autorisées
\end{definition}
~\\

\begin{columns}
\begin{column}[l]{4cm}
Menaces:
\begin{itemize}
\item Ecoute du réseau
\begin{itemize}
\item Interne
\item Externe
\end{itemize}
\item Vol de fichiers 
\begin{itemize}
\item Données
\item Mots de passe
\end{itemize}
\item Espionnage 
\item Ingénierie sociale
\end{itemize}
\end{column}
\begin{column}[r]{6cm}
Contre-mesures:
\begin{itemize}
\item Cryptographie (chiffrement)
\begin{itemize}
\item Des données
\item Des communications
\end{itemize}
\item Contrôle d'accès
\begin{itemize}
\item Logique (mot de passe)
\item Physique (biométrie)
\end{itemize}
\item Classification des actifs
\item Formation du personnel
\end{itemize}
\end{column}
\end{columns}
\end{frame}

\begin{frame}{Intégrité}
\begin{definition}
Propriété d'une donnée dont la valeur est conforme à celle définie
par son propriétaire
\end{definition}
~\\

\begin{columns}
\begin{column}[l]{5cm}
Menaces:
\begin{itemize}
\item Attaques malicieuses
\begin{itemize}
\item Vers, virus
\item Bombes logiques
\end{itemize}
\item Désinformation
\item Erreurs humaines
\end{itemize}
\end{column}
\begin{column}[r]{5cm}
Contre-mesures:
\begin{itemize}
\item Cryptographie
\begin{itemize}
\item Signature, authentification
\end{itemize}
\item Systèmes de détection
\begin{itemize}
\item Antivirus, systèmes de détection d'intrusion (IDS)
\end{itemize}
\item Politique de sauvegarde
\end{itemize}
\end{column}
\end{columns}
\end{frame}

\begin{frame}{Disponibilité}
\begin{definition}
Propriété d'un S.I capable d'assurer ses
fonctions sans interruption, délai ou dégradation, au moment
même où la sollicitation en est faite
\end{definition}
~\\

\begin{columns}
\begin{column}[l]{5cm}
Menaces:
\begin{itemize}
\item Attaques malicieuses
\begin{itemize}
\item Denis de services
\item SPAM
\end{itemize}
\item Attaques accidentelles
\begin{itemize}
\item Le \textit{"Slashdot effect"}
\end{itemize}
\item Pannes
\end{itemize}
\end{column}
\begin{column}[r]{5cm}
Contre-mesures:
\begin{itemize}
\item Pare-feu
\item Systèmes de détection d'intrusions
\item Clustering
\item Formation des administrateurs
\end{itemize}
\end{column}
\end{columns}
\end{frame}

\section{La sécurité en entreprise}

\subsection{Politique de sécurité}

\begin{frame}{Définition}
\begin{definition}
Les politiques de sécurité sont des énoncés généraux dictées
par les cadres supérieurs décrivant le rôle de la sécurité au sein
de l'entreprise afin d'assurer les objectifs d'affaire.
\end{definition}
~\\

\begin{itemize}
\item Pour mettre en oeuvre ces politiques, une organisation doit être
mise en place.
\item Définition des rôles, des responsabilités et des imputabilités
\end{itemize}
\end{frame}

\begin{frame}{Politique de sécurité}
Compromis sécurité - fonctionnalité :
\begin{itemize}
\item Définir les besoins.
\begin{itemize}
\item Déterminer les actifs à protéger et leurs propriétaires
\begin{itemize}
\item Quelles sont leurs valeurs? Quelles sont leurs criticités? Quelles sont leurs propriétés ?
\end{itemize}
\item Déterminer les menaces représentant des risques
\begin{itemize}
\item Quels sont les acteurs ? attaqueurs? Quels sont leurs moyens ?
\end{itemize}
\item Déterminer les objectifs à atteindre
\begin{itemize}
\item Quelles sont les propriétés des actifs à protéger ?
\end{itemize}
\end{itemize} 
\item Proposer une solution.
\begin{itemize}
\item Déterminer les contre-mesures à mettre en place
\end{itemize}
\item Évaluer les risques résiduels.
\begin{itemize}
\item Déterminer quelles sont les vulnérabilités toujours présentes
\item Déterminer leurs impacts sur les objectifs initiaux
\end{itemize}
\end{itemize}
\end{frame}

\begin{frame}{Politique de sécurité (2)}
Création d'une politique de sécurité:
\begin{itemize}
\item Mise en oeuvre
\item Audit
\item Tests d'intrusion
\item Détection d'incidents
\item Réactions
\item Restauration
\end{itemize}
\end{frame}

\begin{frame}{Le problème}
L'industrie de la sécurité
\begin{itemize}
\item  En 2003, le marché de la sécurité réseau était évalué à 45 milliards USD
\item Constat
\begin{itemize}
\item L'approche traditionnelle de sécuriser le périmètre du réseau ne semble
pas adéquate puisque nous avons toujours les mêmes problèmes
\end{itemize}
\item Raisons:
\begin{itemize}
\item Le manque d'information des utilisateurs
\item La \textbf{qualité des logiciels}
\end{itemize}
\end{itemize}
\end{frame}

\begin{frame}{Le rôle du développeur}
\begin{itemize}
\item La sécurité des logiciels est donc critique!
\begin{itemize}
\item 50 \% des vulnérabilités proviennent des erreurs de conception
\item 50 \% des vulnérabilités proviennent des erreurs d'implémentation
\begin{itemize}
\item Dépassement de mémoire et d'entier
\item Concurrence critique
\end{itemize}
\end{itemize}
\item Microsoft's Trustworthy Computing Initiative
\begin{itemize}
\item Mémo de Bill Gates en janvier 2002 présente la nouvelle approche de
Microsoft de développer des logiciels sécurisés
\item Microsoft aurait dépensé plus de 300 millions USD
\end{itemize}
\item Workshop on Rapide Malware 2006
\begin{itemize}
\item Un panel reconnaissait l'impact de cette initiative sur la prévalence des
vers et des virus
\end{itemize}
\end{itemize}
\end{frame}

\begin{frame}{Solution}
Plusieurs solutions:
\begin{itemize}
\item Développement fiable (cycle en $V$, en $W$, etc.)
\item Politique qualité au sein de l'entreprise (ISO, CMMI)
\item \textbf{Informer} !
\end{itemize}
\end{frame}
